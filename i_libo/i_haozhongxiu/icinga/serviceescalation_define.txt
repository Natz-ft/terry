define serviceescalation{
	主机名
	host_name 	

	主机的组名,如果主机名指定将失效
	hostgroup_name 	

	服务组名，如果服务名指定将失效
	servicegroup_name 	

	服务名
	service_description 	

	联系人
	contacts 	

	联系人组名
	contact_groups 	

	首次发送通知的次数
	first_notification 	

	最后发送通知的次数
	last_notification 	

	通知间隔
	notification_interval 	

	该通知扩展的有效时间段
	escalation_period
	
	该通知扩展有效的故障等级
	escalation_options [w,u,c,r]
	
	该通知扩展的有效性条件
	escalation_condition <condition> ( [ & / | ] <condition> )*

	首次发警告通知的次数
	first_warning_notification
	
	最后发送警告通知的次数
	last_warning_notification
	
	首次发紧急通知的次数
	first_critical_notification

	最后发送紧急通知的次数	
	last_critical_notification
	
	首次发未知告警通知的次数
	first_unknown_notification

	最后发送未知告警通知的次数
	last_unknown_notification
